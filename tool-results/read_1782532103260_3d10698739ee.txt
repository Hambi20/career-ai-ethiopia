   304→// ========== AUTO-APPLY TAB (Review & Apply) ==========
   305→function AutoApplyTab({
   306→  applications, setApplications,
   307→  pdfDialogApp, setPdfDialogApp, pdfData, setPdfData, isGeneratingPdf, setIsGeneratingPdf,
   308→  isRunningCycle, setIsRunningCycle, runLogs, setRunLogs,
   309→  expandedCoverLetter, setExpandedCoverLetter,
   310→  countdownStr,
   311→}: {
   312→  applications: Application[]; setApplications: (a: Application[]) => void;
   313→  pdfDialogApp: Application | null; setPdfDialogApp: (a: Application | null) => void;
   314→  pdfData: { cvHtml: string; coverLetterHtml: string } | null; setPdfData: (d: any) => void;
   315→  isGeneratingPdf: boolean; setIsGeneratingPdf: (v: boolean) => void;
   316→  isRunningCycle: boolean; setIsRunningCycle: (v: boolean) => void;
   317→  runLogs: string[]; setRunLogs: (l: string[]) => void;
   318→  expandedCoverLetter: string | null; setExpandedCoverLetter: (id: string | null) => void;
   319→  countdownStr: string;
   320→}) {
   321→  const [isApproving, setIsApproving] = useState<string | null>(null);
   322→
   323→  const refreshApps = useCallback(async () => {
   324→    try {
   325→      const res = await fetch('/api/applications');
   326→      const data = await res.json();
   327→      if (data.success) setApplications(data.applications);
   328→    } catch { /* ignore */ }
   329→  }, [setApplications]);
   330→
   331→  const pendingApps = applications.filter(a => a.status === 'pending_review').sort((a, b) => (b.matchScore || 0) - (a.matchScore || 0));
   332→  const approvedApps = applications.filter(a => a.status === 'approved');
   333→  const submittedApps = applications.filter(a => a.status === 'submitted');
   334→
   335→  // Trigger async search with polling
   336→  const triggerRun = async (category?: string) => {
   337→    if (isRunningCycle) return;
   338→    setIsRunningCycle(true);
   339→    const catLabel = category === 'linkedin' ? 'LinkedIn' : category === 'remote' ? 'Remote Data Entry' : category === 'ethiopia' ? 'Ethiopian Sites' : 'All Sources';
   340→    setRunLogs([`🔍 Starting ${catLabel} job search...`, 'Please wait — results appear here live...']);
   341→    try {
   342→      // Step 1: Start the search (returns immediately)
   343→      const queryStr = category ? `?category=${category}` : '';
   344→      const res = await fetch(`/api/auto-apply/run${queryStr}`, { method: 'POST' });
   345→      const data = await res.json();
   346→      if (!data.success) {
   347→        setRunLogs(['❌ ' + (data.error || 'Failed to start search')]);
   348→        toast.error(data.error || 'Failed to start');
   349→        setIsRunningCycle(false);
   350→        return;
   351→      }
   352→      if (data.alreadyRunning) {
   353→        setRunLogs(['⚠️ A search is already running, polling its progress...']);
   354→      }
   355→      const searchId = data.searchId;
   356→
   357→      // Step 2: Poll for results every 3 seconds
   358→      const poll = setInterval(async () => {
   359→        try {
   360→          const statusRes = await fetch(`/api/auto-apply/run?searchId=${searchId}`);
   361→          const status = await statusRes.json();
   362→          if (status.logs) setRunLogs(status.logs);
   363→          if (status.status === 'completed') {
   364→            clearInterval(poll);
   365→            setIsRunningCycle(false);
   366→            const autoApproved = status.results?.autoApproved || 0;
   367→            const autoSubmitted = status.results?.autoSubmitted || 0;
   368→            const saved = status.results?.totalSaved || 0;
   369→            const found = status.results?.totalFound || 0;
   370→            toast.success(`[${catLabel}] Found ${found} jobs, ${saved} new saved${autoApproved > 0 ? `, ${autoApproved} auto-approved` : ''}${autoSubmitted > 0 ? `, ${autoSubmitted} auto-submitted` : ''}`);
   371→            if (autoSubmitted > 0) {
   372→              setTimeout(() => toast.info(`${autoSubmitted} high-match jobs were auto-submitted with cover letters!`), 1500);
   373→            }
   374→            refreshApps();
   375→          } else if (status.status === 'failed') {
   376→            clearInterval(poll);
   377→            setIsRunningCycle(false);
   378→            toast.error('Search failed: ' + (status.error || 'Unknown error'));
   379→          }
   380→        } catch {
   381→          // Keep polling even if one poll fails
   382→        }
   383→      }, 3000);
   384→    } catch (e) {
   385→      setRunLogs(['❌ ' + String(e)]);
   386→      toast.error('Failed to start search');
   387→      setIsRunningCycle(false);
   388→    }
   389→  };
   390→
   391→  // Approve single
   392→  const approveApp = async (id: string) => {
   393→    setIsApproving(id);
   394→    try {
   395→      const res = await fetch(`/api/applications/${id}/approve`, { method: 'POST' });
   396→      const data = await res.json();
   397→      if (data.success) {
   398→        toast.success('Application approved! Ready to send.');
   399→        refreshApps();
   400→      } else toast.error(data.error || 'Failed');
   401→    } catch { toast.error('Error approving'); }
   402→    finally { setIsApproving(null); }
   403→  };
   404→
   405→  // Reject single
   406→  const rejectApp = async (id: string) => {
   407→    try {
   408→      const res = await fetch(`/api/applications/${id}/reject`, { method: 'POST' });
   409→      const data = await res.json();
   410→      if (data.success) {
   411→        toast.success('Application rejected');
   412→        refreshApps();
   413→      } else toast.error(data.error || 'Failed');
   414→    } catch { toast.error('Error rejecting'); }
   415→  };
   416→
   417→  // Batch approve all pending
   418→  const batchApprove = async () => {
   419→    if (pendingApps.length === 0) return;
   420→    try {
