   248→/**
   249→ * Wraps a promise with an AbortController-based timeout.
   250→ */
   251→function withTimeout<T>(
   252→  promise: Promise<T>,
   253→  ms: number,
   254→  label: string
   255→): Promise<T> {
   256→  return Promise.race([
   257→    promise,
   258→    new Promise<never>((_, reject) =>
   259→      setTimeout(
   260→        () => reject(new Error(`${label} timed out after ${ms}ms`)),
   261→        ms
   262→      )
   263→    ),
   264→  ]);
   265→}

   267→async function searchJobs(query: string): Promise<SearchResult[]> {
   268→  try {
   269→    const zai = await getZAI();
   270→
   271→    // Primary search targeting Ethiopian job sites
   272→    const siteFilter = ETHIOPIAN_JOB_SITES
   273→      .slice(0, 4)
   274→      .map((s) => `site:${s}`)
   275→      .join(" OR ");
   276→    const targetedQuery = `${query} ${siteFilter}`;
   277→
   278→    console.log(`  [web_search] Targeted: "${targetedQuery.substring(0, 80)}..."`);
   279→    const targetedResults = await withTimeout(
   280→      zai.functions.invoke("web_search", {
   281→        query: targetedQuery,
   282→        num: 8,
   283→        recency_days: 30,
   284→      }),
   285→      60_000,
   286→      `web_search(targeted) for "${query.substring(0, 40)}"`
   287→    );
   288→
   289→    // Broader fallback search
   290→    const broadQuery = `latest ${query} 2025`;
   291→    console.log(`  [web_search] Broad: "${broadQuery.substring(0, 80)}..."`);
   292→    const broadResults = await withTimeout(
   293→      zai.functions.invoke("web_search", {
   294→        query: broadQuery,
   295→        num: 5,
   296→        recency_days: 14,
   297→      }),
   298→      60_000,
   299→      `web_search(broad) for "${query.substring(0, 40)}"`
   300→    );
   301→
   302→    // Combine and deduplicate
   303→    const allResults = [
   304→      ...(Array.isArray(targetedResults) ? targetedResults : []),
   305→      ...(Array.isArray(broadResults) ? broadResults : []),
   306→    ];
   307→
   308→    const seenUrls = new Set<string>();
   309→    const filtered: SearchResult[] = [];
   310→
   311→    for (const r of allResults) {
   312→      if (!r || typeof r !== "object") continue;
   313→
   314→      const url = String(r.url || "");
   315→      if (!url) continue;
   316→      if (seenUrls.has(url)) continue;
   317→
   318→      // Skip social media feeds
   319→      if (
   320→        url.includes("facebook.com") ||
   321→        url.includes("twitter.com") ||
   322→        url.includes("linkedin.com/feed") ||
   323→        url.includes("instagram.com")
   324→      ) {
   325→        continue;
   326→      }
   327→
   328→      seenUrls.add(url);
   329→
   330→      const hostName = String(r.host_name || new URL(url).hostname.replace("www.", ""));
   331→      const title = String(r.name || r.title || "");
   332→      const snippet = String(r.snippet || r.description || "");
   333→
   334→      if (!title) continue;
   335→
   336→      filtered.push({
   337→        title,
   338→        snippet,
   339→        url,
   340→        source: ETHIOPIAN_JOB_SITES.find((s) => hostName.includes(s)) || hostName,
   341→        hostName,
   342→        date: r.date ? String(r.date) : undefined,
   343→      });
   344→    }
   345→
   346→    return filtered;
   347→  } catch (err) {
   348→    console.error(`[WebSearch] Error searching "${query}":`, err);
   349→    return [];
   350→  }
   351→}

   353→// ---------------------------------------------------------------------------
   354→// Step 2 — LLM Score Match (0-100)
   355→// ---------------------------------------------------------------------------
   356→
   357→async function scoreMatch(
   358→  job: SearchResult
   359→): Promise<{ score: number; reasoning: string }> {
   360→  const userMessage = `You are a professional recruitment assistant. Rate how well the following job posting matches the candidate's profile.
   361→
   362→Provide ONLY a JSON object with two fields:
   363→- "score": a number from 0 to 100 (higher = better match)
   364→- "reasoning": a brief 1-2 sentence explanation
   365→
   366→${buildProfileSummary()}
   367→
   368→---
   369→
   370→**Job Posting:**
   371→- Title: ${job.title}
   372→- Description: ${job.snippet}
   373→- Source: ${job.source}
   374→- URL: ${job.url}
   375→
   376→Respond with ONLY the JSON object, no markdown fences.`;
   377→
   378→  try {
   379→    const zai = await getZAI();
   380→    console.log(`  [LLM] Scoring ${job.title.substring(0, 60)}...`);
   381→    const completion = await withTimeout(
   382→      zai.chat.completions.create({
   383→        messages: [
   384→          {
   385→            role: "assistant",
   386→            content:
   387→              "You are a JSON-only response assistant. Always respond with valid JSON and nothing else.",
   388→          },
   389→          { role: "user", content: userMessage },
   390→        ],
   391→        thinking: { type: "disabled" },
   392→      }),
   393→      60_000,
   394→      `LLM score for "${job.title.substring(0, 40)}"`
   395→    );
   396→
   397→    const text =
   398→      completion?.choices?.[0]?.message?.content?.trim() || "";
   399→
   400→    // Strip markdown fences if the LLM wraps the JSON
   401→    const cleaned = text
   402→      .replace(/```json?\s*\n?/g, "")
   403→      .replace(/```\s*$/g, "")
   404→      .trim();
   405→
   406→    const parsed = JSON.parse(cleaned);
   407→
   408→    return {
   409→      score: Math.max(0, Math.min(100, Number(parsed.score) || 0)),
   410→      reasoning: String(parsed.reasoning || "No reasoning provided"),
   411→    };
   412→  } catch (err) {
   413→    console.error("[LLM] Score error for", job.title, ":", err);
   414→    return { score: 0, reasoning: "Failed to score — parsing error" };
   415→  }
   416→}

   520→// ---------------------------------------------------------------------------
   521→// Company extraction heuristics
   522→// ---------------------------------------------------------------------------
   523→
   524→function extractCompany(job: SearchResult): string {
   525→  const text = `${job.title} ${job.snippet}`;
   526→  const patterns = [
   527→    /(?:at|@|by|from)\s+([A-Z][A-Za-z&\s]{2,35}?)(?:\s*[-–|,.\n:;]|\s+(?:is|hiring|needs|looking|seeks|wants)|$)/,
   528→    /([A-Z][A-Za-z&\s]{2,35}?)\s+(?:is hiring|needs|looking for|seeks|wants|invites)/i,
   529→    /(?:company|organization|firm)[:\s]*([A-Z][A-Za-z&\s]{2,35}?)(?:\s*[-–|,.\n:;]|$)/i,
   530→  ];
   531→
   532→  for (const pat of patterns) {
   533→    const match = text.match(pat);
   534→    if (match && match[1]) {
   535→      const name = match[1].trim();
   536→      // Filter out common false positives
   537→      const blacklist = new Set([
   538→        "Ethiopia",
   539→        "Addis Ababa",
   540→        "Job",
   541→        "Vacancy",
   542→        "Career",
   543→        "Hiring",
   544→        "View",
   545→        "Apply",
   546→        "Click",
   547→        "Read",
   548→        "More",
   549→        "Details",
   550→        "Full",
   551→        "Description",
   552→      ]);
   553→      if (!blacklist.has(name) && name.length > 2) {
   554→        return name;
   555→      }
   556→    }
   557→  }
   558→
   559→  return job.source;
   560→}

   562→// ---------------------------------------------------------------------------
   563→// Core: Run Full Search Cycle
   564→// ---------------------------------------------------------------------------
   565→
   566→async function runSearchCycle(): Promise<CycleLog> {
   567→  const cycle: CycleLog = {
   568→    cycleId: uid(),
   569→    startedAt: nowISO(),
   570→    totalJobsFound: 0,
   571→    matched: 0,
   572→    saved: 0,
   573→    failed: 0,
   574→    skipped: 0,
   575→    logs: [],
   576→  };
   577→
   578→  currentCycle = cycle;
   579→  isRunning = true;
   580→  console.log(`\n${"=".repeat(64)}`);
   581→  console.log(
   582→    `[Cycle ${cycle.cycleId}] Starting auto-search cycle at ${cycle.startedAt}`
   583→  );
   584→  console.log(`${"=".repeat(64)}\n`);
   585→
   586→  // Track seen URLs across all queries to avoid duplicates
   587→  const seenUrls = new Set<string>();
   588→
   589→  for (const query of SEARCH_QUERIES) {
   590→    console.log(`[Search] Querying: "${query}"`);
   591→    const results = await searchJobs(query);
   592→    console.log(`[Search] Found ${results.length} unique results`);
   593→
   594→    for (const job of results) {
   595→      // Deduplicate globally
   596→      if (seenUrls.has(job.url)) continue;
   597→      seenUrls.add(job.url);
   598→      cycle.totalJobsFound++;
   599→
   600→      console.log(`\n[Evaluate] ${job.title} @ ${job.source}`);
   601→
   602→      // Step 2: Score match with LLM
   603→      const { score, reasoning } = await scoreMatch(job);
   604→      console.log(`[Evaluate] Match score: ${score}/100 — ${reasoning}`);
   605→
   606→      // Extract company name
   607→      const company = extractCompany(job);
   608→
   609→      const logEntry: ApplicationLog = {
   610→        id: uid(),
   611→        timestamp: nowISO(),
   612→        query,
   613→        jobTitle: job.title,
   614→        company,
   615→        source: job.source,
   616→        matchScore: score,
   617→        coverLetter: "",
   618→        status: "skipped",
   619→        url: job.url,
   620→      };
   621→
   622→      // Step 2b: Skip if below threshold
   623→      if (score < MATCH_THRESHOLD) {
   624→        console.log(
   625→          `[Skip] Score ${score} is below threshold ${MATCH_THRESHOLD}`
   626→        );
   627→        cycle.skipped++;
   628→        cycle.logs.push(logEntry);
   629→        continue;
   630→      }
   631→
   632→      // Step 3: Generate cover letter
   633→      cycle.matched++;
   634→      console.log(`[Match] Score ${score} ≥ ${MATCH_THRESHOLD} — generating cover letter...`);
   635→
   636→      const coverLetter = await generateCoverLetter(job, company);
   637→      logEntry.coverLetter = coverLetter;
   638→
   639→      // Step 4: Save to Next.js
   640→      const success = await saveApplication(logEntry);
   641→      logEntry.status = success ? "saved" : "failed";
   642→      if (!success) {
   643→        logEntry.error = "Failed to save to Next.js API";
   644→      }
   645→
   646→      if (success) {
   647→        cycle.saved++;
   648→        console.log(`[Saved] ✓ ${job.title} @ ${company}`);
   649→      } else {
   650→        cycle.failed++;
   651→        console.error(`[Failed] ✗ ${job.title} @ ${company}`);
   652→      }
   653→
   654→      cycle.logs.push(logEntry);
   655→
   656→      // Throttle to avoid API rate limits
   657→      await sleep(2000);
   658→    }
   659→
   660→    // Brief pause between queries
   661→    await sleep(1000);
   662→  }
   663→
   664→  cycle.completedAt = nowISO();
   665→  isRunning = false;
   666→  currentCycle = null;
   667→
   668→  console.log(`\n${"=".repeat(64)}`);
   669→  console.log(
   670→    `[Cycle ${cycle.cycleId}] Complete at ${cycle.completedAt}`
   671→  );
   672→  console.log(
   673→    `  Jobs found : ${cycle.totalJobsFound}\n` +
   674→    `  Matched (≥${MATCH_THRESHOLD}) : ${cycle.matched}\n` +
   675→    `  Saved      : ${cycle.saved}\n` +
   676→    `  Failed     : ${cycle.failed}\n` +
   677→    `  Skipped    : ${cycle.skipped}`
   678→  );
   679→  console.log(`${"=".repeat(64)}\n`);
   680→
   681→  // Store cycle log (keep last 20 in memory)
   682→  cycleLogs.unshift(cycle);
   683→  if (cycleLogs.length > 20) cycleLogs.pop();
   684→
   685→  return cycle;
   686→}

   688→// ---------------------------------------------------------------------------
   689→// HTTP Server
   690→// ---------------------------------------------------------------------------
   691→
   692→const server = Bun.serve({
   693→  port: PORT,
   694→  async fetch(req: Request): Promise<Response> {
   695→    const url = new URL(req.url);
   696→    const path = url.pathname;
   697→
   698→    // CORS headers
   699→    const corsHeaders: Record<string, string> = {
   700→      "Access-Control-Allow-Origin": "*",
   701→      "Access-Control-Allow-Methods": "GET, POST, OPTIONS",
   702→      "Access-Control-Allow-Headers": "Content-Type",
   703→    };
   704→
   705→    // Handle CORS preflight
   706→    if (req.method === "OPTIONS") {
   707→      return new Response(null, { status: 204, headers: corsHeaders });
   708→    }
   709→
   710→    try {
   711→      // ---------------------------------------------------------------
   712→      // POST /api/auto-search — trigger a search cycle
   713→      // ---------------------------------------------------------------
   714→      if (path === "/api/auto-search" && req.method === "POST") {
   715→        if (isRunning) {
   716→          return jsonResponse(
   717→            {
   718→              status: "already_running",
   719→              message:
   720→                "A search cycle is already in progress. Please wait for it to complete.",
   721→              currentCycle: currentCycle
   722→                ? {
   723→                    cycleId: currentCycle.cycleId,
   724→                    startedAt: currentCycle.startedAt,
   725→                    totalJobsFound: currentCycle.totalJobsFound,
   726→                    matched: currentCycle.matched,
   727→                    saved: currentCycle.saved,
   728→                  }
   729→                : null,
   730→            },
   731→            202,
   732→            corsHeaders
   733→          );
   734→        }
   735→
   736→        // Fire-and-forget: start cycle, return immediately
   737→        runSearchCycle().catch((err) => {
   738→          console.error("[Server] Search cycle crashed:", err);
   739→          isRunning = false;
   740→          currentCycle = null;
   741→        });
   742→
   743→        return jsonResponse(
   744→          {
   745→            status: "started",
   746→            message:
   747→              "Auto-search cycle has been triggered. Check GET /api/status for progress.",
   748→          },
   749→          202,
   750→          corsHeaders
   751→        );
   752→      }
   753→
   754→      // ---------------------------------------------------------------
   755→      // GET /api/status — current status + recent application logs
   756→      // ---------------------------------------------------------------
   757→      if (path === "/api/status" && req.method === "GET") {
   758→        const lastCycle = cycleLogs[0] || null;
   759→        const totalSaved = cycleLogs.reduce((s, c) => s + c.saved, 0);
   760→        const totalMatched = cycleLogs.reduce((s, c) => s + c.matched, 0);
   761→        const totalFailed = cycleLogs.reduce((s, c) => s + c.failed, 0);
   762→        const totalSkipped = cycleLogs.reduce((s, c) => s + c.skipped, 0);
   763→        const totalJobs = cycleLogs.reduce((s, c) => s + c.totalJobsFound, 0);
   764→
   765→        // Recent logs from last 2 cycles, capped at 20 entries
   766→        const recentLogs: ApplicationLog[] = cycleLogs
   767→          .slice(0, 2)
   768→          .flatMap((c) => c.logs)
   769→          .slice(0, 20);
   770→
   771→        return jsonResponse(
   772→          {
   773→            service: "auto-apply-service",
   774→            candidate: PROFILE.name,
   775→            port: PORT,
   776→            isRunning,
   777→            currentCycle: currentCycle
   778→              ? {
   779→                  cycleId: currentCycle.cycleId,
   780→                  startedAt: currentCycle.startedAt,
   781→                  totalJobsFound: currentCycle.totalJobsFound,
   782→                  matched: currentCycle.matched,
   783→                  saved: currentCycle.saved,
   784→                }
   785→              : null,
   786→            lastCycle: lastCycle
   787→              ? {
   788→                  cycleId: lastCycle.cycleId,
   789→                  startedAt: lastCycle.startedAt,
   790→                  completedAt: lastCycle.completedAt,
   791→                  totalJobsFound: lastCycle.totalJobsFound,
   792→                  matched: lastCycle.matched,
   793→                  saved: lastCycle.saved,
   794→                  failed: lastCycle.failed,
   795→                  skipped: lastCycle.skipped,
   796→                }
   797→              : null,
   798→            allTimeStats: {
   799→              totalCycles: cycleLogs.length,
   800→              totalJobsFound: totalJobs,
   801→              totalMatched,
   802→              totalSaved,
   803→              totalFailed,
   804→              totalSkipped,
   805→            },
   806→            recentLogs,
   807→          },
   808→          200,
   809→          corsHeaders
   810→        );
   811→      }
   812→
   813→      // ---------------------------------------------------------------
   814→      // GET /api/logs — all application logs with optional filters
   815→      // ---------------------------------------------------------------
   816→      if (path === "/api/logs" && req.method === "GET") {
   817→        const allLogs = cycleLogs.flatMap((c) => c.logs);
   818→        const limitParam = url.searchParams.get("limit");
   819→        const minScoreParam = url.searchParams.get("minScore");
   820→        const statusParam = url.searchParams.get("status");
   821→
   822→        let filtered = allLogs;
   823→
   824→        if (minScoreParam) {
   825→          const threshold = Number(minScoreParam);
   826→          if (!isNaN(threshold)) {
   827→            filtered = filtered.filter((l) => l.matchScore >= threshold);
   828→          }
   829→        }
   830→
   831→        if (statusParam) {
   832→          filtered = filtered.filter((l) => l.status === statusParam);
   833→        }
   834→
   835→        if (limitParam) {
   836→          const n = Number(limitParam);
   837→          if (!isNaN(n) && n > 0) {
   838→            filtered = filtered.slice(0, n);
   839→          }
   840→        }
   841→
   842→        return jsonResponse(
   843→          {
   844→            total: filtered.length,
   845→            cycles: cycleLogs.length,
   846→            logs: filtered,
   847→          },
   848→          200,
   849→          corsHeaders
   850→        );
   851→      }
   852→
   853→      // ---------------------------------------------------------------
   854→      // 404 — Unknown route
   855→      // ---------------------------------------------------------------
   856→      return jsonResponse(
   857→        {
   858→          error: "Not Found",
   859→          availableEndpoints: [
   860→            "POST /api/auto-search  — Trigger a new search cycle",
   861→            "GET  /api/status       — View service status and recent logs",
   862→            "GET  /api/logs         — View all application logs",
   863→            'GET  /api/logs?limit=50&minScore=50&status=saved  — Filtered logs',
   864→          ],
   865→        },
   866→        404,
   867→        corsHeaders
   868→      );
   869→    } catch (err) {
   870→      console.error("[Server] Unhandled error:", err);
   871→      return jsonResponse(
   872→        {
   873→          error: "Internal Server Error",
   874→          message: err instanceof Error ? err.message : "Unknown error",
   875→        },
   876→        500,
   877→        corsHeaders
   878→      );
   879→    }
   880→  },
   881→});

   893→    status,
   894→    headers: {
   895→      "Content-Type": "application/json",
   896→      ...extraHeaders,
   897→    },
   898→  });
   899→}
