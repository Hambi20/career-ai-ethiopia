   221→// ========== APPLICATIONS TAB ==========
   222→function ApplicationsTab({ applications, setApplications }: {
   223→  applications: Application[]; setApplications: (apps: Application[]) => void;
   224→}) {
   225→  const [filterStatus, setFilterStatus] = useState('all');
   226→  const [filterText, setFilterText] = useState('');
   227→  const [expandedId, setExpandedId] = useState<string | null>(null);
   228→  const [showAddDialog, setShowAddDialog] = useState(false);
   229→  const [isUpdating, setIsUpdating] = useState<string | null>(null);
   230→  const [newApp, setNewApp] = useState({ jobTitle: '', company: '', location: '', status: 'preparing', url: '' });
   231→
   232→  const refreshApps = useCallback(async () => {
   233→    try {
   234→      const res = await fetch('/api/applications');
   235→      const data = await res.json();
   236→      if (data.success) setApplications(data.applications);
   237→    } catch { /* ignore */ }
   238→  }, [setApplications]);
   239→
   240→  const filtered = applications.filter((app) => {
   241→    if (filterStatus !== 'all' && app.status !== filterStatus) return false;
   242→    if (filterText) {
   243→      const t = filterText.toLowerCase();
   244→      return app.jobTitle.toLowerCase().includes(t) || (app.company || '').toLowerCase().includes(t);
   245→    }
   246→    return true;
   247→  });
   248→
   249→  const statusCounts = { all: applications.length, 'auto-applied': 0, applied: 0, submitted: 0, preparing: 0, interview: 0, offered: 0, rejected: 0, withdrawn: 0 };
   250→  applications.forEach((a) => { const s = a.status as keyof typeof statusCounts; if (s in statusCounts) statusCounts[s]++; });
   251→
   252→  const updateStatus = async (id: string, status: string) => {
   253→    setIsUpdating(id);
   254→    try {
   255→      await fetch('/api/applications', { method: 'PUT', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ id, status }) });
   256→      toast.success('Status updated');
   257→      refreshApps();
   258→    } catch { toast.error('Failed to update'); }
   259→    finally { setIsUpdating(null); }
   260→  };
   261→
   262→  const deleteApp = async (id: string) => {
   263→    try {
   264→      await fetch(`/api/applications?id=${id}`, { method: 'DELETE' });
   265→      toast.success('Application deleted');
   266→      refreshApps();
   267→    } catch { toast.error('Failed to delete'); }
   268→  };
   269→
   270→  const addApp = async () => {
   271→    if (!newApp.jobTitle.trim()) { toast.error('Job title required'); return; }
   272→    try {
   273→      await fetch('/api/applications', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(newApp) });
   274→      toast.success('Application added');
   275→      setShowAddDialog(false);
   276→      setNewApp({ jobTitle: '', company: '', location: '', status: 'preparing', url: '' });
   277→      refreshApps();
   278→    } catch { toast.error('Failed to add'); }
   279→  };
   280→
   281→  const statusColor = (s: string) => s === 'auto-applied' || s === 'applied' ? 'bg-emerald-100 text-emerald-800 dark:bg-emerald-900/30 dark:text-emerald-300' :
   282→    s === 'submitted' ? 'bg-blue-100 text-blue-800 dark:bg-blue-900/30 dark:text-blue-300' :
   283→    s === 'interview' ? 'bg-purple-100 text-purple-800' :
   284→    s === 'offered' ? 'bg-amber-100 text-amber-800' :
   285→    s === 'rejected' ? 'bg-red-100 text-red-800' :
   286→    s === 'withdrawn' ? 'bg-gray-200 text-gray-600' : 'bg-gray-100 text-gray-800';
   287→
   288→  return (
   289→    <div className="space-y-4">
   290→      {/* Stats */}
   291→      <div className="grid grid-cols-3 sm:grid-cols-5 gap-2">
   292→        {Object.entries(statusCounts).filter(([k]) => k !== 'all' && statusCounts[k as keyof typeof statusCounts] > 0).slice(0, 5).map(([key, count]) => (
   293→          <Card key={key} className={filterStatus === key ? 'border-emerald-500' : ''} onClick={() => setFilterStatus(key === filterStatus ? 'all' : key)}>
   294→            <CardContent className="p-3 text-center cursor-pointer">
   295→              <p className="text-lg font-bold">{count}</p>
   296→              <p className="text-[10px] text-muted-foreground capitalize">{key.replace('-', ' ')}</p>
   297→            </CardContent>
   298→          </Card>
   299→        ))}
   300→      </div>
   301→
   302→      {/* Filters + Add */}
   303→      <div className="flex flex-col sm:flex-row gap-2">
   304→        <Input placeholder="Search by title or company..." value={filterText} onChange={(e) => setFilterText(e.target.value)} className="flex-1 h-9 text-sm" />
   305→        <Select value={filterStatus} onValueChange={setFilterStatus}>
   306→          <SelectTrigger className="w-40 h-9 text-xs"><SelectValue placeholder="All Status" /></SelectTrigger>
   307→          <SelectContent>
   308→            <SelectItem value="all">All Status</SelectItem>
   309→            <SelectItem value="auto-applied">Auto-Applied</SelectItem>
   310→            <SelectItem value="applied">Applied</SelectItem>
   311→            <SelectItem value="submitted">Submitted</SelectItem>
   312→            <SelectItem value="interview">Interview</SelectItem>
   313→            <SelectItem value="offered">Offered</SelectItem>
   314→            <SelectItem value="rejected">Rejected</SelectItem>
   315→            <SelectItem value="withdrawn">Withdrawn</SelectItem>
   316→            <SelectItem value="preparing">Preparing</SelectItem>
   317→          </SelectContent>
   318→        </Select>
   319→        <Button onClick={() => setShowAddDialog(true)} size="sm" className="bg-emerald-600 hover:bg-emerald-700 text-white gap-1.5 h-9"><Plus className="w-3.5 h-3.5" />Add</Button>
   320→      </div>
   321→
   322→      {/* List */}
   323→      <Card>
   324→        <CardContent className="p-3">
   325→          {filtered.length === 0 ? (
   326→            <div className="text-center py-12 text-muted-foreground">
   327→              <FileText className="w-10 h-10 mx-auto mb-2 opacity-30" />
   328→              <p className="text-sm">{applications.length === 0 ? 'No applications yet. Auto-apply will create them.' : 'No applications match filters.'}</p>
   329→            </div>
   330→          ) : (
   331→            <ScrollArea className="max-h-[600px]">
   332→              <div className="space-y-2">
   333→                {filtered.map((app) => (
   334→                  <div key={app.id} className="border rounded-lg overflow-hidden">
   335→                    <div className="flex items-center gap-2.5 p-3 hover:bg-muted/30 transition-colors">
   336→                      <div className="w-8 h-8 rounded-lg bg-emerald-100 dark:bg-emerald-900/30 flex items-center justify-center flex-shrink-0">
   337→                        {app.matchScore && app.matchScore >= 70 ? <Check className="w-4 h-4 text-emerald-600" /> : <Briefcase className="w-4 h-4 text-emerald-600" />}
   338→                      </div>
   339→                      <div className="flex-1 min-w-0">
   340→                        <div className="flex items-center gap-1.5">
   341→                          <p className="text-sm font-medium truncate">{app.jobTitle}</p>
   342→                          {app.matchScore && <Badge variant="outline" className="text-[10px] px-1.5 py-0">{app.matchScore}%</Badge>}
   343→                        </div>
   344→                        <div className="flex items-center gap-1.5 flex-wrap mt-0.5">
   345→                          {app.company && <span className="text-xs text-muted-foreground flex items-center gap-0.5"><Building2 className="w-2.5 h-2.5" />{app.company}</span>}
   346→                          {app.location && <span className="text-xs text-muted-foreground flex items-center gap-0.5"><MapPin className="w-2.5 h-2.5" />{app.location}</span>}
   347→                          {app.source && <span className="text-[10px] text-muted-foreground">{app.source}</span>}
   348→                        </div>
   349→                      </div>
   350→                      <div className="flex items-center gap-1.5 flex-shrink-0">
   351→                        <Badge className={`text-[10px] px-2 py-0 border-0 ${statusColor(app.status)}`}>{app.status}</Badge>
   352→                        <span className="text-[10px] text-muted-foreground hidden sm:inline">{new Date(app.createdAt).toLocaleDateString()}</span>
   353→                        <div className="flex items-center gap-0.5 ml-1">
   354→                          <Select value={app.status} onValueChange={(v) => updateStatus(app.id, v)}>
   355→                            <SelectTrigger className="w-7 h-7 p-0 border-0" disabled={isUpdating === app.id}>
   356→                              {isUpdating === app.id ? <Loader2 className="w-3 h-3 animate-spin" /> : null}
   357→                              <SelectValue />
   358→                            </SelectTrigger>
   359→                            <SelectContent>
   360→                              <SelectItem value="preparing">Preparing</SelectItem>
   361→                              <SelectItem value="auto-applied">Auto-Applied</SelectItem>
   362→                              <SelectItem value="submitted">Submitted</SelectItem>
   363→                              <SelectItem value="interview">Interview</SelectItem>
   364→                              <SelectItem value="offered">Offered</SelectItem>
   365→                              <SelectItem value="rejected">Rejected</SelectItem>
   366→                              <SelectItem value="withdrawn">Withdrawn</SelectItem>
   367→                            </SelectContent>
   368→                          </Select>
   369→                          {app.url && <Button asChild variant="ghost" size="sm" className="h-7 w-7 p-0"><a href={app.url} target="_blank" rel="noopener noreferrer"><ExternalLink className="w-3 h-3" /></a></Button>}
   370→                          {app.coverLetter && <Button variant="ghost" size="sm" className="h-7 w-7 p-0" onClick={() => setExpandedId(expandedId === app.id ? null : app.id)}><Eye className="w-3 h-3" /></Button>}
   371→                          <Button variant="ghost" size="sm" className="h-7 w-7 p-0 text-red-500" onClick={() => deleteApp(app.id)}><Trash2 className="w-3 h-3" /></Button>
   372→                        </div>
   373→                      </div>
   374→                    </div>
   375→                    {expandedId === app.id && app.coverLetter && (
   376→                      <div className="px-3 pb-3 border-t bg-muted/20">
   377→                        <div className="flex items-center justify-between mb-1.5 mt-2">
   378→                          <p className="text-xs font-semibold text-emerald-600">Cover Letter</p>
   379→                          <Button variant="ghost" size="sm" className="h-6 text-xs gap-1" onClick={() => { navigator.clipboard.writeText(app.coverLetter || ''); toast.success('Copied!'); }}><Copy className="w-3 h-3" />Copy</Button>
   380→                        </div>
   381→                        <ScrollArea className="max-h-48"><p className="text-xs whitespace-pre-line leading-relaxed font-serif text-muted-foreground">{app.coverLetter}</p></ScrollArea>
   382→                      </div>
