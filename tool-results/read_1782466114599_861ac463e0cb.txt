   256тЖТ   256тЖТfunction withTimeout<T>(
   257тЖТ   257тЖТ  promise: Promise<T>,
   258тЖТ   258тЖТ  ms: number,
   259тЖТ   259тЖТ  label: string
   260тЖТ   260тЖТ): Promise<T> {
   261тЖТ   261тЖТ  return Promise.race([
   262тЖТ   262тЖТ    promise,

   264тЖТ   264тЖТ      setTimeout(
   265тЖТ   265тЖТ        () => reject(new Error(`${label} timed out after ${ms}ms`)),
   266тЖТ   266тЖТ        ms
   267тЖТ   267тЖТ      )
   268тЖТ   268тЖТ    ),
   269тЖТ   269тЖТ  ]);
