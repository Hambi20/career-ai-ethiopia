    61→function useLocalState() {
    62→  const [currentUser, setCurrentUser] = useState<AuthUser | null>(null);
    63→  const [authToken, setAuthToken] = useState<string | null>(null);
    64→  const [showAuthModal, setShowAuthModal] = useState(false);
    65→  const [authMode, setAuthMode] = useState<'login' | 'register'>('login');
    66→  const [authLoading, setAuthLoading] = useState(false);
    67→  const [authForm, setAuthForm] = useState({ email: '', password: '', name: '', phone: '' });
    68→  const [activeTab, setActiveTab] = useState<Tab>('dashboard');
    69→  const [applications, setApplications] = useState<Application[]>([]);
    70→  const [profile, setProfile] = useState<UserProfile | null>(null);
    71→  const [chatMessages, setChatMessages] = useState<ChatMessage[]>([]);
    72→  const [isChatLoading, setIsChatLoading] = useState(false);
    73→  const [chatContext, setChatContext] = useState('job-search');
    74→  const [message, setMessage] = useState('');
    75→  const [generatedCoverLetter, setGeneratedCoverLetter] = useState<string | null>(null);
    76→  const [isGeneratingCoverLetter, setIsGeneratingCoverLetter] = useState(false);
    77→  const [coverLetterForm, setCoverLetterForm] = useState({ jobTitle: '', company: '', jobDescription: '' });
    78→  const [expandedCoverLetter, setExpandedCoverLetter] = useState<string | null>(null);
    79→  const [pdfDialogApp, setPdfDialogApp] = useState<Application | null>(null);
    80→  const [pdfData, setPdfData] = useState<{ cvHtml: string; coverLetterHtml: string } | null>(null);
    81→  const [isGeneratingPdf, setIsGeneratingPdf] = useState(false);
    82→  const [runLogs, setRunLogs] = useState<string[]>([]);
    83→  const [isRunningCycle, setIsRunningCycle] = useState(false);
    84→  const [nextSearchIn, setNextSearchIn] = useState(3600000);
    85→  const [cvAnalysis, setCvAnalysis] = useState<any>(null);
    86→  const [isAnalyzingCv, setIsAnalyzingCv] = useState(false);
    87→  const [interviewPreps, setInterviewPreps] = useState<any[]>([]);
    88→  const [currentPrep, setCurrentPrep] = useState<any>(null);
    89→  const [isGeneratingPrep, setIsGeneratingPrep] = useState(false);
    90→  const [prepForm, setPrepForm] = useState({ jobTitle: '', company: '', jobDescription: '' });
    91→  const [expandedQuestion, setExpandedQuestion] = useState<number | null>(null);
    92→  const [profileEditing, setProfileEditing] = useState(false);
    93→  const [employerJobs, setEmployerJobs] = useState<any[]>([]);
    94→  const chatEndRef = useRef<HTMLDivElement>(null);
    95→
    96→  useEffect(() => {
    97→    const saved = localStorage.getItem('career-ai-token');
    98→    if (saved) {
    99→      setAuthToken(saved);
   100→      fetch('/api/auth/me', { headers: { Authorization: 'Bearer ' + saved } })
   101→        .then(r => r.json())
   102→        .then(d => { if (d.success) setCurrentUser(d.user); else localStorage.removeItem('career-ai-token'); })
   103→        .catch(() => localStorage.removeItem('career-ai-token'));
   104→    }
   105→  }, []);
   106→
   107→  const handleLogin = async () => {
   108→    if (!authForm.email || !authForm.password) { toast.error('Email and password required'); return; }
   109→    setAuthLoading(true);
   110→    try {
   111→      const res = await fetch('/api/auth/login', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(authForm) });
   112→      const data = await res.json();
   113→      if (data.success) {
   114→        setCurrentUser(data.user); setAuthToken(data.token);
   115→        localStorage.setItem('career-ai-token', data.token);
   116→        setShowAuthModal(false); setAuthForm({ email: '', password: '', name: '', phone: '' });
   117→        toast.success('Welcome, ' + (data.user.name || data.user.email) + '!');
   118→      } else toast.error(data.error || 'Login failed');
   119→    } catch { toast.error('Login failed'); } finally { setAuthLoading(false); }
   120→  };
   121→
   122→  const handleRegister = async () => {
   123→    if (!authForm.email || !authForm.password || !authForm.name) { toast.error('Name, email and password required'); return; }
   124→    if (authForm.password.length < 6) { toast.error('Password must be 6+ characters'); return; }
   125→    setAuthLoading(true);
   126→    try {
   127→      const res = await fetch('/api/auth/register', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(authForm) });
   128→      const data = await res.json();
   129→      if (data.success) {
   130→        setCurrentUser(data.user); setAuthToken(data.token);
   131→        localStorage.setItem('career-ai-token', data.token);
   132→        setShowAuthModal(false); setAuthForm({ email: '', password: '', name: '', phone: '' });
   133→        toast.success('Account created! Welcome!');
   134→      } else toast.error(data.error || 'Registration failed');
   135→    } catch { toast.error('Registration failed'); } finally { setAuthLoading(false); }
   136→  };
   137→
   138→  const handleLogout = () => { setCurrentUser(null); setAuthToken(null); localStorage.removeItem('career-ai-token'); toast.success('Logged out'); };
   139→
   140→  useEffect(() => { const i = setInterval(() => setNextSearchIn(p => Math.max(0, p - 10000)), 10000); return () => clearInterval(i); }, []);
   141→  useEffect(() => {
   142→    if (nextSearchIn <= 0 && !isRunningCycle) {
   143→      setNextSearchIn(3600000);
   144→      fetch('/api/auto-apply/run?full=true', { method: 'POST' })
   145→        .then(r => r.json()).then(d => { if (d.success) { toast.success('Auto-search found jobs!'); fetch('/api/applications').then(r => r.json()).then(dd => { if (dd.success) setApplications(dd.applications); }); } }).catch(() => {});
   146→      setIsRunningCycle(true); setTimeout(() => setIsRunningCycle(false), 120000);
   147→    }
   148→  }, [nextSearchIn, isRunningCycle]);
   149→  useEffect(() => { const i = setInterval(() => fetch('/api/applications').then(r => r.json()).then(d => { if (d.success) setApplications(d.applications); }).catch(() => {}), 120000); return () => clearInterval(i); }, []);
   150→
   151→  const countdownStr = useMemo(() => Math.floor(nextSearchIn / 60000) + 'm ' + Math.floor((nextSearchIn % 60000) / 1000) + 's', [nextSearchIn]);
   152→
   153→  return {
   154→    currentUser, authToken, showAuthModal, setShowAuthModal, authMode, setAuthMode,
   155→    authLoading, authForm, setAuthForm, handleLogin, handleRegister, handleLogout,
   156→    activeTab, setActiveTab, applications, setApplications, profile, setProfile,
   157→    chatMessages, setChatMessages, isChatLoading, setIsChatLoading, chatContext, setChatContext,
   158→    message, setMessage, generatedCoverLetter, setGeneratedCoverLetter,
   159→    isGeneratingCoverLetter, setIsGeneratingCoverLetter, coverLetterForm, setCoverLetterForm,
   160→    expandedCoverLetter, setExpandedCoverLetter, pdfDialogApp, setPdfDialogApp,
   161→    pdfData, setPdfData, isGeneratingPdf, setIsGeneratingPdf, runLogs, setRunLogs,
   162→    isRunningCycle, setIsRunningCycle, countdownStr, chatEndRef,
   163→    cvAnalysis, setCvAnalysis, isAnalyzingCv, setIsAnalyzingCv,
   164→    interviewPreps, setInterviewPreps, currentPrep, setCurrentPrep,
   165→    isGeneratingPrep, setIsGeneratingPrep, prepForm, setPrepForm,
   166→    expandedQuestion, setExpandedQuestion, profileEditing, setProfileEditing,
   167→    employerJobs, setEmployerJobs,
   168→  };
   169→}

   409→      }, 3000);

   878→function EmployerPortalTab({ employerJobs, setEmployerJobs, currentUser, authToken }: any) {
   879→  const [showForm, setShowForm] = useState(false);
   880→  const [form, setForm] = useState({ title: '', company: '', location: '', type: '', salary: '', category: '', description: '', requirements: '', deadline: '' });
   881→  const [posting, setPosting] = useState(false);
   882→  const headers: Record<string, string> = { 'Content-Type': 'application/json' };
   883→  if (authToken) headers['Authorization'] = 'Bearer ' + authToken;
   884→
   885→  useEffect(() => { fetch('/api/employer/jobs', { headers: { Authorization: 'Bearer ' + authToken } }).then(r => r.json()).then(d => { if (d.success) setEmployerJobs(d.jobs); }); }, [authToken]);
   886→  const handlePost = () => { if (!form.title || !form.description) return; setPosting(true); fetch('/api/employer/jobs', { method: 'POST', headers, body: JSON.stringify(form) }).then(r => r.json()).then(d => { if (d.success) { toast.success('Posted!'); setForm({ title: '', company: '', location: '', type: '', salary: '', category: '', description: '', requirements: '', deadline: '' }); setShowForm(false); fetch('/api/employer/jobs', { headers: { Authorization: 'Bearer ' + authToken } }).then(r => r.json()).then(dd => { if (dd.success) setEmployerJobs(dd.jobs); }); } }); setPosting(false); };
   887→  const handleDel = (id: string) => { fetch('/api/employer/jobs?id=' + id, { method: 'DELETE', headers }); toast.success('Deleted'); setEmployerJobs(employerJobs.filter((j: any) => j.id !== id)); };
   888→
   889→  return (
   890→    <div className="space-y-6">
   891→      <div className="flex items-center justify-between"><h2 className="text-lg font-bold">Employer Portal</h2><Button onClick={() => setShowForm(!showForm)} className="bg-emerald-600 text-white gap-2"><Plus className="w-4 h-4" />Post Job</Button></div>

   897→        <div className="flex gap-2"><Button onClick={handlePost} disabled={posting} className="bg-emerald-600 text-white gap-2">{posting ? <Loader2 className="w-4 h-4 animate-spin" /> : null}{posting ? 'Posting...' : 'Post Job'}</Button><Button variant="outline" onClick={() => setShowForm(false)}>Cancel</Button></div>
   898→      </CardContent></Card>}
   899→      <Card><CardHeader className="pb-3"><CardTitle className="text-base">Your Jobs ({employerJobs.length})</CardTitle></CardHeader><CardContent>
   900→        {employerJobs.length === 0 ? <div className="text-center py-8 text-muted-foreground"><p className="text-sm">No jobs posted</p></div> : (
   901→          <ScrollArea className="max-h-[400px]"><div className="space-y-2">{employerJobs.map((job: any) => (<div key={job.id} className="flex items-center gap-3 p-3 border rounded-lg"><div className="flex-1"><p className="text-sm font-medium">{job.title}</p><p className="text-xs text-muted-foreground">{job.company} • {job.location || 'N/A'}</p></div><Button variant="ghost" size="sm" className="h-8 text-xs text-red-500" onClick={() => handleDel(job.id)}><Trash2 className="w-3.5 h-3.5" /></Button></div>))}</div></ScrollArea>
   902→        )}

   909→function AdminPanelTab({ authToken }: any) {
   910→  const [stats, setStats] = useState<any>(null);
   911→  const [users, setUsers] = useState<any[]>([]);
   912→  const load = () => { const h: Record<string, string> = {}; if (authToken) h['Authorization'] = 'Bearer ' + authToken; Promise.all([fetch('/api/admin/stats', { headers: h }), fetch('/api/admin/users?limit=10', { headers: h })]).then(([sr, ur]) => Promise.all([sr.json(), ur.json()])).then(([sd, ud]) => { if (sd.success) setStats(sd.stats); if (ud.success) setUsers(ud.users); }); };
   913→  useEffect(() => { load(); }, [authToken]);
   914→  const updateUser = (userId: string, updates: any) => { const h: Record<string, string> = { 'Content-Type': 'application/json' }; if (authToken) h['Authorization'] = 'Bearer ' + authToken; fetch('/api/admin/users', { method: 'PUT', headers: h, body: JSON.stringify({ userId, ...updates }) }).then(() => { toast.success('Updated'); load(); }); };
   915→  if (!stats) return <div className="flex justify-center py-12"><Loader2 className="w-8 h-8 animate-spin text-emerald-600" /></div>;
   916→  return (
   917→    <div className="space-y-6">
   918→      <h2 className="text-lg font-bold">Admin Dashboard</h2>
   919→      <div className="grid grid-cols-2 sm:grid-cols-5 gap-3">
   920→        {[{ l: 'Users', v: stats.totalUsers }, { l: 'Seekers', v: stats.totalJobSeekers }, { l: 'Employers', v: stats.totalEmployers }, { l: 'Premium', v: stats.totalPremium }, { l: 'Active Today', v: stats.activeToday }].map(s => <Card key={s.l}><CardContent className="p-4 text-center"><p className="text-xl font-bold">{s.v}</p><p className="text-[10px] text-muted-foreground">{s.l}</p></CardContent></Card>)}

   931→          </div>
