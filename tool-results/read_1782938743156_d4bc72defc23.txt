    19→async function sendTelegramMessage(chatId: number, text: string, options?: {
    20→  parseMode?: string;
    21→  replyMarkup?: any;
    22→  replyToMessageId?: number;
    23→}): Promise<boolean> {
    24→  try {
    25→    const token = getBotToken();
    26→    const chunks: string[] = [];
    27→    const maxLen = 4096;
    28→    if (text.length <= maxLen) {
    29→      chunks.push(text);
    30→    } else {
    31→      // Split on double newlines or single newlines
    32→      const lines = text.split('\n');
    33→      let current = '';
    34→      for (const line of lines) {
    35→        if ((current + '\n' + line).length > maxLen && current.length > 0) {
    36→          chunks.push(current);
    37→          current = line;
    38→        } else {
    39→          current = current ? current + '\n' + line : line;
    40→        }
    41→      }
    42→      if (current) chunks.push(current);
    43→    }
    44→    for (const chunk of chunks) {
    45→      const body: any = {
    46→        chat_id: chatId,
    47→        text: chunk,
    48→        parse_mode: options?.parseMode || 'HTML',
    49→      };
    50→      if (options?.replyMarkup) body.reply_markup = options.replyMarkup;
    51→      if (options?.replyToMessageId) body.reply_to_message_id = options.replyToMessageId;
    52→      const res = await fetch(`${TELEGRAM_API_BASE}/bot${token}/sendMessage`, {
    53→        method: 'POST',
    54→        headers: { 'Content-Type': 'application/json' },
    55→        body: JSON.stringify(body),
    56→      });
    57→      if (!res.ok) console.error('sendMessage failed:', await res.text());
    58→    }
    59→    return true;
    60→  } catch (err) {
    61→    console.error('Send message error:', err);
    62→    return false;
    63→  }
    64→}

    80→  return text.slice(idx + cmd.length).trim();

    99→async function askGroq(prompt: string, systemPrompt?: string): Promise<string> {
   100→  if (!GROQ_API_KEY) return 'AI not configured (GROQ_API_KEY missing).';
   101→  try {
   102→    const messages: any[] = [];
   103→    if (systemPrompt) messages.push({ role: 'system', content: systemPrompt });
   104→    messages.push({ role: 'user', content: prompt });
   105→    const res = await fetch('https://api.groq.com/openai/v1/chat/completions', {
   106→      method: 'POST',
   107→      headers: {
   108→        'Authorization': `Bearer ${GROQ_API_KEY}`,
   109→        'Content-Type': 'application/json',
   110→      },
   111→      body: JSON.stringify({ model: GROQ_MODEL, messages, temperature: 0.7, max_tokens: 2048 }),
   112→    });
   113→    const data = await res.json();
   114→    return data.choices?.[0]?.message?.content || 'AI returned no response.';
   115→  } catch (err: any) {
   116→    console.error('Groq error:', err?.message);
   117→    return `AI error: ${err?.message || 'Unknown'}`;
   118→  }
   119→}

   151→async function saveReportToWeb(
   152→  type: string,
   153→  company: string,
   154→  title: string,
   155→  content: string,
   156→  chatId?: number,
   157→  firstName?: string,
   158→): Promise<boolean> {
   159→  const reportId = uid('rpt');
   160→  const now = nowStr();
   161→  const date = todayStr();
   162→  const esc = (s: string) => (s || '').replace(/'/g, "''").replace(/\n/g, ' ');
   163→  const reportData = { id: reportId, type, company, category: type, title, content, summary: '', chatId: chatId || 0, firstName: firstName || '', date, timestamp: now, createdAt: now };
   164→
   165→  // PRIMARY: Use the db Proxy from @/lib/db
   166→  try {
   167→    await ensureBotReportTable();
   168→    await db.$executeRawUnsafe(
   169→      `INSERT INTO "BotReport" ("id","type","company","category","title","content","summary","chatId","firstName","date","timestamp","createdAt") VALUES ('${esc(reportId)}','${esc(type)}','${esc(company)}','${esc(type)}','${esc(title)}','${esc(content)}','',${chatId || 0},'${esc(firstName || '')}','${date}','${now}',CURRENT_TIMESTAMP)`
   170→    );
   171→    console.log(`[saveReportToWeb] SUCCESS: id=${reportId}, type=${type}`);
   172→    return true;
   173→  } catch (err: any) {
   174→    console.error(`[saveReportToWeb] DB FAILED: ${err?.message}`);
   175→  }
   176→
   177→  // FALLBACK: HTTP POST to production API
   178→  try {
   179→    const res = await fetch(`${APP_URL}/api/bot/reports`, {
   180→      method: 'POST',
   181→      headers: { 'Content-Type': 'application/json' },
   182→      body: JSON.stringify(reportData),
   183→    });
   184→    if (res.ok) {
   185→      console.log(`[saveReportToWeb] HTTP fallback SUCCESS: id=${reportId}`);
   186→      return true;
   187→    }
   188→  } catch (e) { /* silent */ }
   189→
   190→  // LAST RESORT: In-memory
   191→  fallbackReports.unshift(reportData);
   192→  if (fallbackReports.length > 200) fallbackReports.length = 200;
   193→  console.log(`[saveReportToWeb] In-memory fallback: id=${reportId}`);
   194→  return false;
   195→}

  1189→async function handleScrape(chatId: number, args: string) {
  1190→  if (!args) {
  1191→    await sendTelegramMessage(chatId, '<b>🔍 Scrape</b>\n\nUsage: <code>/scrape [URL]</code>\n\nExtracts text content from a web page.');
  1192→    return;
  1193→  }
  1194→  await sendChatAction(chatId, 'typing');
  1195→
  1196→  // Extract URL from args
  1197→  const urlMatch = args.match(/https?:\/\/[^\s]+/);
  1198→  if (!urlMatch) {
  1199→    await sendTelegramMessage(chatId, 'Please provide a valid URL starting with http:// or https://');
  1200→    return;
  1201→  }
  1202→  const url = urlMatch[0];
  1203→
  1204→  try {
  1205→    const res = await fetch(url, {
  1206→      headers: { 'User-Agent': 'Mozilla/5.0 (compatible; HambisaBot/1.0)' },
  1207→      signal: AbortSignal.timeout(15000),
  1208→    });
  1209→    if (!res.ok) throw new Error(`HTTP ${res.status}`);
  1210→    const html = await res.text();
  1211→
  1212→    // Extract text content (simple HTML strip)
  1213→    const text = html
  1214→      .replace(/<script[^>]*>[\s\S]*?<\/script>/gi, '')
  1215→      .replace(/<style[^>]*>[\s\S]*?<\/style>/gi, '')
  1216→      .replace(/<[^>]+>/g, ' ')
  1217→      .replace(/\s+/g, ' ')
  1218→      .trim()
  1219→      .slice(0, 3500);
  1220→
  1221→    const saved = await saveReportToWeb('scrape', '', `Scraped - ${url.slice(0, 60)}`, text, chatId);
  1222→    const tag = saved ? '✅ Synced to dashboard' : '⚠️ Saved locally';
  1223→
  1224→    await sendTelegramMessage(chatId, `<b>🔍 Scraped: ${url.slice(0, 60)}</b>\n\n${text}...\n\n<i>${tag}</i>`);
  1225→  } catch (err: any) {
  1226→    await sendTelegramMessage(chatId, `<b>❌ Scrape Failed</b>\n\n${err?.message || 'Could not fetch URL'}`);
  1227→  }
  1228→}

  1436→      });
