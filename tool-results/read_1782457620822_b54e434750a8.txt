   264→// ========== AUTO-APPLY TAB ==========
   265→function AutoApplyTab({
   266→  autoStatus, autoLogs, isTriggeringSearch, setIsTriggeringSearch,
   267→  applications, expandedCoverLetter, setExpandedCoverLetter
   268→}: {
   269→  autoStatus: AutoApplyStatus | null; autoLogs: Array<Record<string, unknown>>;
   270→  isTriggeringSearch: boolean; setIsTriggeringSearch: (v: boolean) => void;
   271→  applications: Application[]; expandedCoverLetter: string | null; setExpandedCoverLetter: (id: string | null) => void;
   272→}) {
   273→  const [logs, setLocalLogs] = useState(autoLogs);
   274→
   275→  const fetchLogs = useCallback(async () => {
   276→    try {
   277→      const res = await fetch('/api/auto-apply/logs');
   278→      const data = await res.json();
   279→      if (data.success && data.logs) setLocalLogs(data.logs);
   280→    } catch { /* ignore */ }
   281→  }, []);
   282→
   283→  const triggerSearch = async () => {
   284→    setIsTriggeringSearch(true);
   285→    try {
   286→      const res = await fetch('/api/auto-apply/status', { method: 'POST' });
   287→      const data = await res.json();
   288→      if (data.success) toast.success('Auto-search triggered! It will run in background.');
   289→      else toast.error(data.error || 'Failed to trigger');
   290→    } catch { toast.error('Service not reachable'); }
   291→    finally {
   292→      setTimeout(() => setIsTriggeringSearch(false), 2000);
   293→      setTimeout(fetchLogs, 30000);
   294→    }
   295→  };
   296→
   297→  const statusIcon = !autoStatus?.serviceRunning ? <XCircle className="w-5 h-5 text-red-500" /> :
   298→    autoStatus?.status === 'idle' ? <CheckCircle2 className="w-5 h-5 text-emerald-500" /> :
   299→    <Loader2 className="w-5 h-5 text-orange-500 animate-spin" />;
   300→
   301→  const statusText = !autoStatus?.serviceRunning ? 'Service Offline' :
   302→    autoStatus?.status === 'idle' ? 'Ready & Waiting' :
   303→    `Running: ${autoStatus?.currentCycle?.phase || 'Searching...'}`;
   304→
   305→  return (
   306→    <div className="space-y-6">
   307→      {/* Status Card */}
   308→      <Card className="border-emerald-200 dark:border-emerald-800">
   309→        <CardContent className="p-6">
   310→          <div className="flex items-center justify-between mb-4">
   311→            <div className="flex items-center gap-3">
   312→              {statusIcon}
   313→              <div>
   314→                <h3 className="font-semibold">Auto-Apply Service</h3>
   315→                <p className="text-sm text-muted-foreground">{statusText}</p>
   316→              </div>
   317→            </div>
   318→            <Button onClick={triggerSearch} disabled={isTriggeringSearch || !autoStatus?.serviceRunning || autoStatus?.status !== 'idle'} className="bg-emerald-600 hover:bg-emerald-700 text-white gap-2">
   319→              {isTriggeringSearch ? <Loader2 className="w-4 h-4 animate-spin" /> : <Play className="w-4 h-4" />}
   320→              {isTriggeringSearch ? 'Starting...' : 'Run Now'}
   321→            </Button>
   322→          </div>
   323→
   324→          {autoStatus?.stats && (
   325→            <div className="grid grid-cols-3 gap-4">
   326→              <div className="text-center p-3 rounded-lg bg-muted/50">
   327→                <p className="text-2xl font-bold text-emerald-600">{autoStatus.stats.totalJobsFound}</p>
   328→                <p className="text-xs text-muted-foreground">Jobs Found</p>
   329→              </div>
   330→              <div className="text-center p-3 rounded-lg bg-muted/50">
   331→                <p className="text-2xl font-bold text-blue-600">{autoStatus.stats.totalJobsApplied}</p>
   332→                <p className="text-xs text-muted-foreground">Applications Sent</p>
   333→              </div>
   334→              <div className="text-center p-3 rounded-lg bg-muted/50">
   335→                <p className="text-2xl font-bold text-orange-600">{autoStatus.stats.totalCycles}</p>
   336→                <p className="text-xs text-muted-foreground">Search Cycles</p>
   337→              </div>
   338→            </div>
   339→          )}
   340→
   341→          {autoStatus?.lastCycle && (
   342→            <div className="mt-4 p-3 rounded-lg bg-muted/30 border">
   343→              <p className="text-xs text-muted-foreground mb-1">Last Cycle #{autoStatus.lastCycle.cycleNumber} — {autoStatus.lastCycle.completedAt}</p>
   344→              <div className="flex gap-4 text-xs">
   345→                <span>Found: <strong>{autoStatus.lastCycle.jobsFound}</strong></span>
   346→                <span>Matched: <strong>{autoStatus.lastCycle.jobsMatched}</strong></span>
   347→                <span>Applied: <strong>{autoStatus.lastCycle.jobsApplied}</strong></span>
   348→                <span>Duration: <strong>{autoStatus.lastCycle.duration}</strong></span>
   349→              </div>
   350→            </div>
   351→          )}
   352→        </CardContent>
   353→      </Card>
   354→
   355→      {/* Schedule Info */}
   356→      <Card>
   357→        <CardHeader className="pb-3">
   358→          <CardTitle className="text-base flex items-center gap-2"><Radar className="w-4 h-4 text-emerald-600" /> How It Works</CardTitle>
   359→        </CardHeader>
   360→        <CardContent>
   361→          <div className="grid sm:grid-cols-2 gap-4">
   362→            {[
   363→              { step: '1', title: 'Auto-Search', desc: 'Searches 8+ queries across Ethiopian job sites every 4 hours' },
   364→              { step: '2', title: 'Smart Matching', desc: 'LLM scores each job 0-100 against your CV skills & experience' },
   365→              { step: '3', title: 'Cover Letters', desc: 'Generates tailored cover letter for jobs matching 50+ score' },

   367→            ].map((item) => (
   368→              <div key={item.step} className="flex gap-3 p-3 rounded-lg bg-muted/30">
   369→                <div className="w-8 h-8 rounded-full bg-emerald-100 dark:bg-emerald-900/30 flex items-center justify-center text-emerald-600 font-bold text-sm flex-shrink-0">{item.step}</div>

   382→        </CardContent>
   383→      </Card>
   384→
   385→      {/* Application Log */}
   386→      <Card>
   387→        <CardHeader className="pb-3">
   388→          <div className="flex items-center justify-between">
   389→            <CardTitle className="text-base flex items-center gap-2"><ScrollArea className="w-4 h-4 text-emerald-600" /> Application Log</CardTitle>

   391→          </div>
   392→        </CardHeader>
   393→        <CardContent>
   394→          {logs.length === 0 ? (

   397→            <ScrollArea className="max-h-[500px]">
   398→              <div className="space-y-2">
   399→                {logs.map((log, i) => {

   401→                  return (

   607→function AIChatTab({ chatMessages, setChatMessages, isChatLoading, setIsChatLoading, chatContext, setChatContext, message, setMessage, chatEndRef, profile }: {
   608→  chatMessages: ChatMessage[]; setChatMessages: (m: ChatMessage[]) => void;
   609→  isChatLoading: boolean; setIsChatLoading: (v: boolean) => void;
   610→  chatContext: string; setChatContext: (c: string) => void;
   611→  message: string; setMessage: (m: string) => void; chatEndRef: React.RefObject<HTMLDivElement>;
   612→  profile: UserProfile | null;
   613→}) {
   614→  useEffect(() => { chatEndRef.current?.scrollIntoView({ behavior: 'smooth' }); }, [chatMessages, chatEndRef]);
   615→
   616→  const handleSend = async () => {
   617→    if (!message.trim() || isChatLoading) return;
   618→    const userMsg = message.trim();
   619→    setMessage('');
   620→    setIsChatLoading(true);
   621→    const updated = [...chatMessages, { id: `msg-${Date.now()}`, role: 'user' as const, content: userMsg, createdAt: new Date().toISOString() }];
   622→    setChatMessages(updated);
   623→    try {
   624→      const res = await fetch('/api/ai/chat', {
   625→        method: 'POST', headers: { 'Content-Type': 'application/json' },
   626→        body: JSON.stringify({ message: userMsg, context: chatContext }),
   627→      });
   628→      const data = await res.json();
   629→      if (data.success) setChatMessages([...updated, { id: `msg-${Date.now() + 1}`, role: 'assistant' as const, content: data.response, createdAt: new Date().toISOString() }]);
   630→    } catch { toast.error('Failed to get response'); }
   631→    finally { setIsChatLoading(false); }
   632→  };
   633→
   634→  const quickQs = [
   635→    'What are the best sales manager roles available now?',
   636→    'How should I prepare for a sales interview in Ethiopia?',
   637→    'What salary should I expect as a sales manager in Addis Ababa?',
   638→    'How to write a great CV for Ethiopian employers?',
   639→  ];
   640→
   641→  return (
   642→    <div className="space-y-4">
   643→      <div className="flex items-center gap-3 mb-2">
   644→        <div className="w-10 h-10 rounded-full bg-gradient-to-br from-emerald-500 to-teal-600 flex items-center justify-center"><Bot className="w-5 h-5 text-white" /></div>

   650→          <SelectTrigger className="w-32 h-8 text-xs"><SelectValue /></SelectTrigger>

   652→            <SelectItem value="job-search">Job Search</SelectItem>

   654→            <SelectItem value="career">Career</SelectItem>
   655→          </SelectContent>
   656→        </Select>
   657→      </div>
   658→      <Card className="overflow-hidden">
   659→        <div className="h-[400px] flex flex-col">
   660→          <ScrollArea className="flex-1 p-4">
   661→            {chatMessages.length === 0 ? (
   662→              <div className="flex flex-col items-center justify-center h-full text-center">
   663→                <Bot className="w-14 h-14 text-muted-foreground/20 mb-3" />
   664→                <h4 className="font-semibold text-sm mb-1">Ask Your AI Career Coach</h4>

   668→                    <Badge key={q} variant="outline" className="cursor-pointer hover:bg-emerald-50 hover:text-emerald-700 dark:hover:bg-emerald-900 dark:hover:text-emerald-300 text-[10px] py-1 px-2.5 max-w-[220px]" onClick={() => setMessage(q)}>{q}</Badge>
   669→                  ))}
   670→                </div>
   671→              </div>
   672→            ) : (
   673→              <div className="space-y-3">
   674→                {chatMessages.map((msg) => (
   675→                  <div key={msg.id} className={`flex gap-2.5 ${msg.role === 'user' ? 'flex-row-reverse' : ''}`}>
   676→                    <div className={`w-7 h-7 rounded-full flex items-center justify-center flex-shrink-0 ${msg.role === 'user' ? 'bg-primary text-primary-foreground' : 'bg-emerald-100 dark:bg-emerald-900/30 text-emerald-600'}`}>
   677→                      {msg.role === 'user' ? <User className="w-3.5 h-3.5" /> : <Bot className="w-3.5 h-3.5" />}
   678→                    </div>
   679→                    <div className={`max-w-[80%] rounded-lg p-2.5 text-sm ${msg.role === 'user' ? 'bg-primary text-primary-foreground' : 'bg-muted'}`}>
   680→                      <p className="whitespace-pre-line text-xs">{msg.content}</p>

   682→                  </div>

   685→                  <div className="flex gap-2.5"><div className="w-7 h-7 rounded-full bg-emerald-100 dark:bg-emerald-900/30 flex items-center justify-center"><Bot className="w-3.5 h-3.5 text-emerald-600" /></div><div className="bg-muted rounded-lg p-2.5"><div className="flex gap-1"><div className="w-1.5 h-1.5 bg-muted-foreground/40 rounded-full animate-bounce" /><div className="w-1.5 h-1.5 bg-muted-foreground/40 rounded-full animate-bounce" style={{ animationDelay: '150ms' }} /><div className="w-1.5 h-1.5 bg-muted-foreground/40 rounded-full animate-bounce" style={{ animationDelay: '300ms' }} /></div></div></div>
