   265→   265→function AutoApplyTab({
   266→   266→  autoStatus, autoLogs, isTriggeringSearch, setIsTriggeringSearch,
   267→   267→  applications, expandedCoverLetter, setExpandedCoverLetter
   268→   268→}: {
   269→   269→  autoStatus: AutoApplyStatus | null; autoLogs: Array<Record<string, unknown>>;
   270→   270→  isTriggeringSearch: boolean; setIsTriggeringSearch: (v: boolean) => void;
   271→   271→  applications: Application[]; expandedCoverLetter: string | null; setExpandedCoverLetter: (id: string | null) => void;
   272→   272→}) {
   273→   273→  const [logs, setLocalLogs] = useState(autoLogs);
   274→   274→
   275→   275→  const fetchLogs = useCallback(async () => {
   276→   276→    try {
   277→   277→      const res = await fetch('/api/auto-apply/logs');
   278→   278→      const data = await res.json();
   279→   279→      if (data.success && data.logs) setLocalLogs(data.logs);
   280→   280→    } catch { /* ignore */ }
   281→   281→  }, []);
   282→   282→
   283→   283→  const triggerSearch = async () => {
   284→   284→    setIsTriggeringSearch(true);
   285→   285→    try {
   286→   286→      const res = await fetch('/api/auto-apply/status', { method: 'POST' });
   287→   287→      const data = await res.json();
   288→   288→      if (data.success) toast.success('Auto-search triggered! It will run in background.');
   289→   289→      else toast.error(data.error || 'Failed to trigger');
   290→   290→    } catch { toast.error('Service not reachable'); }
   291→   291→    finally {
   292→   292→      setTimeout(() => setIsTriggeringSearch(false), 2000);
   293→   293→      setTimeout(fetchLogs, 30000);
   294→   294→    }
   295→   295→  };
   296→   296→
   297→   297→  const statusIcon = !autoStatus?.serviceRunning ? <XCircle className="w-5 h-5 text-red-500" /> :
   298→   298→    autoStatus?.status === 'idle' ? <CheckCircle2 className="w-5 h-5 text-emerald-500" /> :
   299→   299→    <Loader2 className="w-5 h-5 text-orange-500 animate-spin" />;
   300→   300→
   301→   301→  const statusText = !autoStatus?.serviceRunning ? 'Service Offline' :
   302→   302→    autoStatus?.status === 'idle' ? 'Ready & Waiting' :
   303→   303→    `Running: ${autoStatus?.currentCycle?.phase || 'Searching...'}`;
   304→   304→
   305→   305→  return (
   306→   306→    <div className="space-y-6">
   307→   307→      {/* Status Card */}
   308→   308→      <Card className="border-emerald-200 dark:border-emerald-800">
   309→   309→        <CardContent className="p-6">
   310→   310→          <div className="flex items-center justify-between mb-4">
   311→   311→            <div className="flex items-center gap-3">
   312→   312→              {statusIcon}
   313→   313→              <div>
   314→   314→                <h3 className="font-semibold">Auto-Apply Service</h3>
   315→   315→                <p className="text-sm text-muted-foreground">{statusText}</p>
   316→   316→              </div>
   317→   317→            </div>
   318→   318→            <Button onClick={triggerSearch} disabled={isTriggeringSearch || !autoStatus?.serviceRunning || autoStatus?.status !== 'idle'} className="bg-emerald-600 hover:bg-emerald-700 text-white gap-2">
   319→   319→              {isTriggeringSearch ? <Loader2 className="w-4 h-4 animate-spin" /> : <Play className="w-4 h-4" />}
   320→   320→              {isTriggeringSearch ? 'Starting...' : 'Run Now'}
   321→   321→            </Button>
   322→   322→          </div>
   323→   323→
   324→   324→          {autoStatus?.stats && (
   325→   325→            <div className="grid grid-cols-3 gap-4">
   326→   326→              <div className="text-center p-3 rounded-lg bg-muted/50">
   327→   327→                <p className="text-2xl font-bold text-emerald-600">{autoStatus.stats.totalJobsFound}</p>
   328→   328→                <p className="text-xs text-muted-foreground">Jobs Found</p>
   329→   329→              </div>
   330→   330→              <div className="text-center p-3 rounded-lg bg-muted/50">
   331→   331→                <p className="text-2xl font-bold text-blue-600">{autoStatus.stats.totalJobsApplied}</p>
   332→   332→                <p className="text-xs text-muted-foreground">Applications Sent</p>
   333→   333→              </div>
   334→   334→              <div className="text-center p-3 rounded-lg bg-muted/50">
   335→   335→                <p className="text-2xl font-bold text-orange-600">{autoStatus.stats.totalCycles}</p>
   336→   336→                <p className="text-xs text-muted-foreground">Search Cycles</p>
   337→   337→              </div>
   338→   338→            </div>
   339→   339→          )}
   340→   340→
   341→   341→          {autoStatus?.lastCycle && (
   342→   342→            <div className="mt-4 p-3 rounded-lg bg-muted/30 border">
   343→   343→              <p className="text-xs text-muted-foreground mb-1">Last Cycle #{autoStatus.lastCycle.cycleNumber} — {autoStatus.lastCycle.completedAt}</p>
   344→   344→              <div className="flex gap-4 text-xs">
   345→   345→                <span>Found: <strong>{autoStatus.lastCycle.jobsFound}</strong></span>
   346→   346→                <span>Matched: <strong>{autoStatus.lastCycle.jobsMatched}</strong></span>
   347→   347→                <span>Applied: <strong>{autoStatus.lastCycle.jobsApplied}</strong></span>
   348→   348→                <span>Duration: <strong>{autoStatus.lastCycle.duration}</strong></span>
   349→   349→              </div>
   350→   350→            </div>
   351→   351→          )}
   352→   352→        </CardContent>
